576inline void dRHSdp_getcoeffs(const PetscInt dim, const int it, const int n, const int np, const int i, const int ip, const int stridei, const int strideip, const double* xptr, double* res_p_re, double* res_p_im, double* res_q_re, double* res_q_im, double* res_f_re, double* res_f_im) {
577
578 *res_p_re = 0.0;
579 *res_p_im = 0.0;
580 *res_q_re = 0.0;
581 *res_q_im = 0.0;
582 *res_f_re = 0.0;
583 *res_f_im = 0.0;
584
585 // Flux channel: f(t) multiplies number operator a^\dagger a => diagonal (i-ip) contribution.
586 double xre0 = xptr[it];
587 double xim0 = xptr[it + dim];
588 double coeff = static_cast<double>(i - ip);
589 *res_f_re += coeff * xim0;
590 *res_f_im += -coeff * xre0;
591
592 /* ik+1..,ik'.. term */
593 if (i < n-1) {
594 int itx = it + stridei;
595 double xre = xptr[itx];
596 double xim = xptr[itx+dim];
597 double sq = sqrt(i + 1);
598 *res_p_re += sq * xim;
599 *res_p_im += - sq * xre;
600 *res_q_re += sq * xre;
601 *res_q_im += sq * xim;
602 }
603 /* \rho(ik..,ik'+1..) */
604 if (ip < np-1) {
605 int itx = it + strideip;
606 double xre = xptr[itx];
607 double xim = xptr[itx+dim];
608 double sq = sqrt(ip + 1);
609 *res_p_re += - sq * xim;
610 *res_p_im += + sq * xre;
611 *res_q_re += sq * xre;
612 *res_q_im += sq * xim;
613 }
614 /* \rho(ik-1..,ik'..) */
615 if (i > 0) {
616 int itx = it - stridei;
617 double xre = xptr[itx];
618 double xim = xptr[itx+dim];
619 double sq = sqrt(i);
620 *res_p_re += + sq * xim;
621 *res_p_im += - sq * xre;
622 *res_q_re += - sq * xre;
623 *res_q_im += - sq * xim;
624 }
625 /* \rho(ik..,ik'-1..) */
626 if (ip > 0) {
627 int itx = it - strideip;
628 double xre = xptr[itx];
629 double xim = xptr[itx+dim];
630 double sq = sqrt(ip);
631 *res_p_re += - sq * xim;
632 *res_p_im += + sq * xre;
633 *res_q_re += - sq * xre;
634 *res_q_im += - sq * xim;

664inline void Jkl_coupling(const PetscInt dim, const int it, const int ni, const int nj, const int nip, const int njp, const int i, const int ip, const int j, const int jp, const int stridei, const int strideip, const int stridej, const int stridejp, const double* xptr, const double Jij, const double cosij, const double sinij, double* yre, double* yim) {
665 if (fabs(Jij)>1e-10) {
666 // 1) J_kl (-icos + sin) * ρ_{E−k+l i, i′}
667 if (i > 0 && j < nj-1) {
668 int itx = it - stridei + stridej;
669 double xre = xptr[itx];
670 double xim = xptr[itx+dim];
671 double sq = sqrt(i * (j + 1));
672 // sin u + cos v + i ( -cos u + sin v)
673 *yre += Jij * sq * ( cosij * xim + sinij * xre);
674 *yim += Jij * sq * ( - cosij * xre + sinij * xim);
675 }
676 // 2) J_kl (−icos − sin)sqrt(il*(ik +1)) ρ_{E+k−li,i′}
677 if (i < ni-1 && j > 0) {
678 int itx = it + stridei - stridej; // E+k-l i, i'
679 double xre = xptr[itx];
680 double xim = xptr[itx+dim];
681 double sq = sqrt(j * (i + 1)); // sqrt( il*(ik+1))
682 // -sin u + cos v + i (-cos u - sin v)
683 *yre += Jij * sq * ( cosij * xim - sinij * xre);
684 *yim += Jij * sq * ( - cosij * xre - sinij * xim);
685 }
686 // 3) J_kl ( icos + sin)sqrt(ik'*(il' +1)) ρ_{i,E-k+li'}
687 if (ip > 0 && jp < njp-1) {
688 int itx = it - strideip + stridejp; // i, E-k+l i'
689 double xre = xptr[itx];
690 double xim = xptr[itx+dim];
691 double sq = sqrt(ip * (jp + 1)); // sqrt( ik'*(il'+1))
692 // sin u - cos v + i ( cos u + sin v)
693 *yre += Jij * sq * ( - cosij * xim + sinij * xre);
694 *yim += Jij * sq * ( cosij * xre + sinij * xim);
695 }
696 // 4) J_kl ( icos - sin)sqrt(il'*(ik' +1)) ρ_{i,E+k-li'}
697 if (ip < nip-1 && jp > 0) {
698 int itx = it + strideip - stridejp; // i, E+k-l i'
699 double xre = xptr[itx];
700 double xim = xptr[itx+dim];
701 double sq = sqrt(jp * (ip + 1)); // sqrt( il'*(ik'+1))
702 // - sin u - cos v + i ( cos u - sin v)
703 *yre += Jij * sq * ( - cosij * xim - sinij * xre);
704 *yim += Jij * sq * ( cosij * xre - sinij * xim);
705 }
706 }
